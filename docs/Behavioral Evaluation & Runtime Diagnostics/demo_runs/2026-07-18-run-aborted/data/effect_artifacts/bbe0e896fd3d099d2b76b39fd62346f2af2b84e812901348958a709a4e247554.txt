import math
def gcd(a, b):
    while b:
        a, b = b, a % b
    return a

# a hand-written sort agrees with sorted()
xs = [((i * 13 + 29) % 100) for i in range(11)]
def isort(seq):
    s = list(seq)
    for i in range(1, len(s)):
        k, j = s[i], i - 1
        while j >= 0 and s[j] > k:
            s[j + 1] = s[j]; j -= 1
        s[j + 1] = k
    return s
assert isort(xs) == sorted(xs)
