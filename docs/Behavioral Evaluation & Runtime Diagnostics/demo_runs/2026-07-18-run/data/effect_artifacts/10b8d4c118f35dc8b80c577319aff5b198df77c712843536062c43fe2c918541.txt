import math
n = 22
# closed form vs brute force: sum(1..n) == n(n+1)/2, sum of squares too
assert sum(range(1, n + 1)) == n * (n + 1) // 2
assert sum(i * i for i in range(1, n + 1)) == n * (n + 1) * (2 * n + 1) // 6
# gcd/lcm identity
a, b = n, 10
g = math.gcd(a, b)
assert g * (a * b // g) == a * b
print("CHECK_PASS arithmetic identities hold for n =", n)
