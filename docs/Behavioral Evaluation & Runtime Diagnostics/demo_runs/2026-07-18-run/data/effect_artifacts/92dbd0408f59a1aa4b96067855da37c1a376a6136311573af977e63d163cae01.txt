# a hand-written sort agrees with sorted()
xs = [((i * 13 + 7) % 100) for i in range(12)]
def isort(seq):
    s = list(seq)
    for i in range(1, len(s)):
        k, j = s[i], i - 1
        while j >= 0 and s[j] > k:
            s[j + 1] = s[j]; j -= 1
        s[j + 1] = k
    return s
assert isort(xs) == sorted(xs)
